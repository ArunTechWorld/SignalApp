  declarations: [
    LoginComponent,
    focusinput,
    ForgotPasswordComponent,
  ],
  imports: [
    HttpClientModule,
    IonicPageModule.forChild(LoginComponent),
  ],
  providers: [
    LoginProvider,
  ],
  entryComponents: [
    ForgotPasswordComponent,
  ],
})
export class LoginPageModule { }
